def checkio(array: list[int]) -> int:

    if len(array)!= 0:
        sumEven = sum([array[i] for i in range(0, len(array), 2)])
        last = array[len(array) - 1]
        return sumEven * last 
    else:
        return 0

def checkio(array: list[int]) -> int:
            
    try:
        return array[-1] * sum(array[::2]) 
    except IndexError:
        return 0
